### New Classification Functions
```javascript
// pH-specific classification
function classifyPh(value) {
  if (value < 5.5) return { status: "red", text: "HIGH ACIDIC" };
  else if (value > 8.5) return { status: "red", text: "HIGHLY ALKALINE" };
  else return { status: "green", text: "NEUTRAL" };
}

// EC-specific classification
function classifyEc(value, max) {
  if (value <= max) return { status: "green", text: "NORMAL" };
  else return { status: "red", text: "HIGH SALINE" };
}
```
